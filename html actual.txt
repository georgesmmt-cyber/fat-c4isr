(function() {
    'use strict';
    
    console.log('☁️ Initialisation sync cloud...');
    
    window.cloudSync = {
        enabled: true,
        
        async sauvegarderHistorique(actions) {
            try {
                const user = window.currentUser?.name || 'GeorgesD';
                const date = new Date().toISOString().split('T')[0];
                
                await db.collection('historique').doc(`${user}_${date}`).set({
                    user: user,
                    date: date,
                    actions: actions || [],
                    total: actions?.length || 0,
                    lastUpdate: new Date().toISOString()
                });
                
                console.log('☁️ Historique sauvegardé');
                return true;
                
            } catch (error) {
                console.log('⚠️ Erreur sync:', error);
                return false;
            }
        },
        
        async synchroniser() {
            console.log('☁️ Synchronisation...');
            
            if (window.actionHistory?.actions) {
                return await this.sauvegarderHistorique(window.actionHistory.actions);
            } else {
                // Créer un historique minimal s'il n'existe pas
                if (!window.actionHistory) {
                    window.actionHistory = { actions: [], save: () => {} };
                }
                
                // Ajouter une action de test
                window.actionHistory.actions.push({
                    id: `act_${Date.now()}`,
                    description: 'Première sync cloud',
                    type: 'system',
                    timestamp: new Date().toISOString(),
                    user: 'GeorgesD'
                });
                
                return await this.sauvegarderHistorique(window.actionHistory.actions);
            }
        }
    };
    
    // Synchronisation automatique au démarrage
    setTimeout(() => {
        window.cloudSync.synchroniser();
        // Puis toutes les heures
        setInterval(() => window.cloudSync.synchroniser(), 3600000);
    }, 5000);
    
    console.log('✅ Sync cloud prête');
    console.log('📌 Commande: cloudSync.synchroniser()');
    
})();

function ajouterBoutonSync() {
    // Attendre que la top bar existe
    const interval = setInterval(() => {
        const topBar = document.querySelector('.top-bar');
        if (!topBar) return;
        
        // Éviter les doublons
        if (document.getElementById('sync-cloud-btn')) {
            clearInterval(interval);
            return;
        }
        
        // Créer le bouton
        const btnSync = document.createElement('button');
        btnSync.id = 'sync-cloud-btn';
        btnSync.className = 'env-btn';
        btnSync.style.cssText = 'margin-left: 10px; background: #0066cc;';
        btnSync.innerHTML = '<i class="fas fa-cloud-upload-alt"></i> SYNC';
        
        btnSync.onclick = async () => {
            btnSync.disabled = true;
            btnSync.innerHTML = '<i class="fas fa-spinner fa-spin"></i> SYNC...';
            
            await window.cloudSync.synchroniser();
            
            btnSync.disabled = false;
            btnSync.innerHTML = '<i class="fas fa-cloud-upload-alt"></i> SYNC';
            
            alert('✅ Synchronisation terminée');
        };
        
        topBar.appendChild(btnSync);
        console.log('✅ Bouton SYNC ajouté à la top bar');
        clearInterval(interval);
        
    }, 500); // Vérifier toutes les 500ms
}

            // Sauf pour Esc qui ferme
            if (e.key !== 'Escape') return;
        }
        
        // Construire la combinaison
        const keys = [];
        if (e.ctrlKey) keys.push('Ctrl');
        if (e.shiftKey) keys.push('Shift');
        if (e.altKey) keys.push('Alt');
        
        // Gérer les chiffres
        if (e.key >= '0' && e.key <= '9') {
            keys.push(e.key);
        } else {
            keys.push(e.key.length === 1 ? e.key.toUpperCase() : e.key);
        }
        
        const combo = keys.join('+');

function updateTabsVisibility() {

    // Si pas de rôle défini, forcer ADMIN
    if (!currentUserRole) {
        currentUserRole = 'ADMIN';

    }
    
    document.querySelectorAll('.tab-item').forEach(tab => {
        const tabName = tab.dataset.tab;
        
        // Admin voit tout
        if (currentUserRole === 'ADMIN') {
            tab.style.display = 'flex';
        } 
        // Autres rôles
        else {
            const role = ROLES[currentUserRole] || ROLES.VISITOR;
            if (role.canAccess.includes(tabName)) {
                tab.style.display = 'flex';
            } else {
                tab.style.display = 'none';
            }
        }
    });
    
    // Forcer l'affichage du chat
    const chatTab = document.querySelector('[data-tab="chat"]');
    if (chatTab) {
        chatTab.style.display = 'flex';
        console.log('✅ Chat tab forcé');
    }
    
    console.log('✅ Onglets mis à jour');
}

        function playMilitaryBeep(freq, duration) {
            if (!audioEnabled || !audioCtx || audioCtx.state === 'closed') return;
            if (audioCtx.state === 'suspended') { audioCtx.resume(); return; }
            try {
                const osc = audioCtx.createOscillator();
                const gain = audioCtx.createGain();
                osc.type = 'square';
                osc.frequency.setValueAtTime(freq, audioCtx.currentTime);
                gain.gain.setValueAtTime(0.1, audioCtx.currentTime);
                osc.connect(gain).connect(audioCtx.destination);
                osc.start();
                osc.stop(audioCtx.currentTime + duration);
            } catch(e) {}

    function optimizeModel(model, gender) {
        model.traverse((node) => {
            if (node.isMesh) {
                node.castShadow = false;
                node.receiveShadow = false;
                
                if (node.material.map) {
                    node.material.map.anisotropy = 1;
                    node.material.map.minFilter = THREE.LinearFilter;
                    node.material.map.magFilter = THREE.LinearFilter;
                    node.material.map.generateMipmaps = false;
                }
                
                const materials = Array.isArray(node.material) ? node.material : [node.material];
                materials.forEach(mat => {
                    mat.precision = 'lowp';
                    mat.needsUpdate = true;
                    
                    if (gender === 'male') {
                        mat.emissive = new THREE.Color(0x112233);
                        mat.emissiveIntensity = 0.3;
                        mat.metalness = 0.3;
                        mat.roughness = 0.4;
                    } else {
                        mat.emissive = new THREE.Color(0x221133);
                        mat.emissiveIntensity = 0.3;
                        mat.metalness = 0.2;
                        mat.roughness = 0.5;
                    }
                });
            }
        });
        return model;
    }

       const scaleFactor = gender === 'female' ? 0.35 : 0.9;

        model.position.z = -center.z * scaleFactor;
        
        console.log(`📏 Modèle ${gender}: échelle=${scaleFactor}`);
        return model;
    }

let dataInitialized = false;

function connectWebSocket() {
    if (wsConnection && wsConnection.readyState === WebSocket.OPEN) return;
    
    wsConnection = new WebSocket('ws://localhost:5001');
    
    wsConnection.onopen = () => {
        console.log('✅ WebSocket connecté - mode événementiel actif');
        
        // Une seule fois : charger les données initiales
        if (!dataInitialized) {
            console.log('📦 Chargement initial des données');
            loadAllData(true);
            dataInitialized = true;
        }
        
        // S'abonner aux événements
        wsConnection.send(JSON.stringify({ type: 'subscribe_all' }));
    };
    
    wsConnection.onmessage = (event) => {
        try {
            const data = JSON.parse(event.data);
            
            // Traiter uniquement les changements
            switch(data.type) {
                case 'data_changed':
                    // Recharger SEULEMENT les données modifiées
                    if (data.table === 'vests') {
                        console.log('🔄 Mise à jour unités');
                        refreshVestsOnly();
                    } else if (data.table === 'alerts') {
                        console.log('🔔 Nouvelle alerte');
                        refreshAlertsOnly();
                    } else if (data.table === 'positions') {
                        console.log('📍 Mise à jour positions');
                        refreshPositionsOnly();
                    }
                    break;
                    
                case 'new_alert':
                    console.log('🚨 Alerte temps réel');
                    addAlertToUI(data.alert);
                    break;
                    
                case 'position_update':
                    console.log('📍 Position mise à jour');
                    updateMarkerOnMap(data.position);
                    break;
                    
                default:
                    // Ignorer les autres messages
                    break;
            }
        } catch (e) {
            console.log('Message WebSocket ignoré');
        }
    };
    
    wsConnection.onclose = () => {
        console.log('🔌 WebSocket déconnecté');
        setTimeout(connectWebSocket, 5000);
    };
}

async function loadAllData(force = false) {
    if (isLoadingData) return;
    isLoadingData = true;
    
    try {
        const [vests, positions, alerts, dashboard, summary] = await Promise.all([
            api.getVests().catch(() => []),
            api.getLatestPositions().catch(() => []),
            api.getPendingAlerts().catch(() => []),
            api.getDashboardData().catch(() => ({ stats: [] })),
            api.getAnalyticsSummary().catch(() => [])
        ]);
        
        updateUnitsDisplay(vests);
        updatePositionsOnMap(positions);
        updateAlertsDisplay(alerts);
        updateDashboardWithData(dashboard);
        updateSummary(summary);
        
        console.log('✅ Données initiales chargées');
        
    } catch (error) {
        console.error('❌ Erreur chargement initial:', error);
    } finally {
        isLoadingData = false;
    }
}

// Fonction pour fermer proprement la connexion
function disconnectWebSocket() {
    if (wsPingInterval) {
        clearInterval(wsPingInterval);
        wsPingInterval = null;
    }
    
    if (wsReconnectTimeout) {
        clearTimeout(wsReconnectTimeout);
        wsReconnectTimeout = null;
    }
    
    if (wsConnection) {
        try {
            wsConnection.close(1000, "Fermeture normale");
        } catch (e) {}
        wsConnection = null;
    }
    
    wsReconnectAttempts = 0;
    console.log('🔌 WebSocket déconnecté manuellement');
}

// CHARGEMENT SÉQUENTIEL AVEC COMPTEUR CORRIGÉ
// =====================================================
let consecutiveErrors = 0;
let isLoading = false;
let lastSuccessfulLoad = 0;
const MAX_ERRORS = 3;
const MIN_LOAD_INTERVAL = 30000; // 30 secondes

async function loadAllData(force = false) {
    const now = Date.now();
    
    // Vérifier le verrou
    if (isLoading) {
        console.log('⏳ Chargement déjà en cours, ignoré');
        return;
    }
    
    // Vérifier l'intervalle minimum (sauf force)
    if (!force && lastSuccessfulLoad > 0 && (now - lastSuccessfulLoad) < MIN_LOAD_INTERVAL) {
        console.log(`⏳ Prochain chargement dans ${Math.ceil((MIN_LOAD_INTERVAL - (now - lastSuccessfulLoad))/1000)}s`);
        return;
    }
    
    // Après trop d'erreurs, pause prolongée
    if (consecutiveErrors >= MAX_ERRORS && !force) {
        const pauseTime = Math.min(60000, consecutiveErrors * 10000);
        console.log(`⚠️ Trop d'erreurs (${consecutiveErrors}), pause de ${pauseTime/1000}s`);
        setTimeout(() => loadAllData(true), pauseTime);
        return;
    }
    
    isLoading = true;
    
    // Compteurs pour savoir si au moins UNE requête a réussi
    let anySuccess = false;
    let successCount = 0;
    const totalRequests = 5;
    
    try {
        console.log('📊 Chargement séquentiel des données...');
        
        // 1. D'abord les unités (critique)
        let vests = [];
        try {
            vests = await api.getVests();
            if (vests && vests.length > 0) anySuccess = true;
            console.log(`✅ ${vests.length} unités chargées`);
            successCount++;
        } catch (e) {
            console.warn('⚠️ Échec vests:', e.message);
        }
        updateUnitsDisplay(vests);
        
        // 2. Ensuite les positions
        let positions = [];
        try {
            positions = await api.getLatestPositions();
            if (positions && positions.length > 0) anySuccess = true;
            console.log(`✅ ${positions.length} positions chargées`);
            successCount++;
        } catch (e) {
            console.warn('⚠️ Échec positions:', e.message);
        }
        updatePositionsOnMap(positions);
        
        // 3. Puis les alertes
        let alerts = [];
        try {
            alerts = await api.getPendingAlerts();
            if (alerts && alerts.length > 0) anySuccess = true;
            console.log(`✅ ${alerts.length} alertes chargées`);
            successCount++;
        } catch (e) {
            console.warn('⚠️ Échec alertes:', e.message);
        }
        updateAlertsDisplay(alerts);
        
        // 4. Dashboard
        let dashboard = { stats: [] };
        try {
            dashboard = await api.getDashboardData();
            if (dashboard && (dashboard.units?.length > 0 || dashboard.stats?.length > 0)) anySuccess = true;
            console.log(`✅ Dashboard chargé`);
            successCount++;
        } catch (e) {
            console.warn('⚠️ Échec dashboard:', e.message);
        }
        updateDashboardWithData(dashboard);
        
        // 5. Enfin summary
        let summary = [];
        try {
            summary = await api.getAnalyticsSummary();
            if (summary && summary.length > 0) anySuccess = true;
            console.log(`✅ Summary chargé`);
            successCount++;
        } catch (e) {
            console.warn('⚠️ Échec summary:', e.message);
        }
        updateSummary(summary);
        
        // ✅ CORRECTION CRUCIALE : Réinitialiser le compteur SI au moins une requête a réussi
        if (anySuccess) {
            consecutiveErrors = 0;
            lastSuccessfulLoad = Date.now();
            console.log(`✅ Chargement réussi (${successCount}/${totalRequests} requêtes OK)`);
        } else {
            // Aucune requête n'a réussi
            consecutiveErrors++;
            console.error(`❌ Toutes les requêtes ont échoué (${consecutiveErrors}/${MAX_ERRORS})`);
        }
        
    } catch (error) {
        consecutiveErrors++;
        console.error(`❌ Erreur chargement (${consecutiveErrors}/${MAX_ERRORS}):`, error.message);
        
        // Afficher un message dans l'interface
        if (typeof showErrorMessage === 'function') {
            showErrorMessage('Problème de connexion au serveur - réessai automatique', 'warning');
        }
        
    } finally {
        isLoading = false;
    }
}

        function updatePositionsOnMap(positions) {
            if (!missionMap) return;
            
            missionMap.eachLayer((layer) => {
                if (layer instanceof L.Marker || layer instanceof L.CircleMarker) {

                        fillColor: color,

                    `);
                    
                    marker.on('click', () => showBiometric(pos.id));

// MONITORING AMÉLIORÉ - AVEC CACHE
// =====================================================
let sensorCache = {
    data: null,
    timestamp: 0
};
const CACHE_DURATION = 10000; // 10 secondes

// =====================================================
// CHARGEMENT DES DONNÉES CAPTEURS - AVEC GESTION D'ERREURS
// =====================================================
async function loadLatestSensorData() {
    // Vérifier le cache
    const now = Date.now();
    if (sensorCache.data && (now - sensorCache.timestamp) < CACHE_DURATION) {
        console.log('📦 Utilisation du cache capteurs');
        displayMonitoringStats(sensorCache.data);
        return;
    }
    
    try {
        const vests = await api.getVests();
        const monitoringContent = document.getElementById('monitoring-content');
        
        const latestReadings = [];
        
        // Limiter à 3 unités pour réduire la charge
        const unitsToLoad = vests.slice(0, 3);
        
        for (const vest of unitsToLoad) {
            try {
                // ÉTAPE 5: Gestion des erreurs de colonne "time"
                let readings;
                try {
                    readings = await api.getSensorReadings(vest.id, 1);
                } catch (apiError) {
                    // Vérifier si c'est une erreur de colonne "time"
                    if (apiError.message && apiError.message.includes('column "time" does not exist')) {
                        console.log(`⚠️ Erreur de colonne time pour ${vest.id} - utilisation données simulées`);
                        
                        // Simuler des données pour ce vest
                        latestReadings.push({
                            sensor_type: `VEST ${vest.id} (simulé)`,
                            avg_value: Math.random() * 10 + 35, // Température entre 35-45
                            unit: '°C'
                        });
                        continue;
                    } else {
                        // Autre erreur API
                        throw apiError;
                    }
                }
                
                if (readings && readings.length > 0) {
                    const reading = readings[0];
                    if (reading.data) {
                        latestReadings.push({
                            sensor_type: `VEST ${vest.id}`,
                            avg_value: reading.data.temperature || reading.data.heart_rate || reading.data.battery || '--',
                            unit: reading.data.temperature ? '°C' : 
                                  reading.data.heart_rate ? 'BPM' : 
                                  reading.data.battery ? '%' : ''
                        });
                    } else if (reading.temperature || reading.heart_rate || reading.battery) {
                        // Format alternatif si data n'existe pas
                        latestReadings.push({
                            sensor_type: `VEST ${vest.id}`,
                            avg_value: reading.temperature || reading.heart_rate || reading.battery || '--',
                            unit: reading.temperature ? '°C' : 
                                  reading.heart_rate ? 'BPM' : 
                                  reading.battery ? '%' : ''
                        });
                    }
                }
            } catch (e) {
                console.log(`⚠️ Capteur ${vest.id} non disponible:`, e.message);
                
                // ÉTAPE 5: Fallback - données simulées pour ce capteur
                latestReadings.push({
                    sensor_type: `VEST ${vest.id} (fallback)`,
                    avg_value: Math.random() * 10 + 35,
                    unit: '°C'
                });
            }
        }
        
        let statsToDisplay;
        if (latestReadings.length > 0) {
            statsToDisplay = latestReadings;
        } else {
            statsToDisplay = getSimulatedStats();
        }
        
        // Mettre en cache
        sensorCache.data = statsToDisplay;
        sensorCache.timestamp = now;
        
        displayMonitoringStats(statsToDisplay);
        
    } catch (error) {
        console.error('❌ Erreur chargement capteurs:', error);
        
        // Utiliser le cache même expiré en cas d'erreur
        if (sensorCache.data) {
            displayMonitoringStats(sensorCache.data);
        } else {
            displayMonitoringStats(getSimulatedStats());
        }
    }
}

// Fonction changePassword
window.changePassword = async function() {
    const current = document.getElementById('current-password')?.value;
    const newPass = document.getElementById('new-password')?.value;
    const confirm = document.getElementById('confirm-password')?.value;
    const messageEl = document.getElementById('password-message');
    
    if (!messageEl) return;
    
    // Validation
    if (!current || !newPass || !confirm) {
        messageEl.innerHTML = '<span style="color:#ff4444;">❌ Tous les champs sont requis</span>';
        return;
    }
    
    if (newPass.length < 8) {
        messageEl.innerHTML = '<span style="color:#ff4444;">❌ Le mot de passe doit contenir au moins 8 caractères</span>';
        return;
    }
    
    if (newPass !== confirm) {
        messageEl.innerHTML = '<span style="color:#ff4444;">❌ Les mots de passe ne correspondent pas</span>';
        return;
    }
    
    try {
        const result = await api.request('/auth/change-password', {
            method: 'POST',
            body: JSON.stringify({ 
                currentPassword: current, 
                newPassword: newPass 
            })
        });
        
        messageEl.innerHTML = '<span style="color:#0f0;">✅ Mot de passe modifié avec succès</span>';
        
        setTimeout(() => {
            closeModal('password-modal');
            // Réinitialiser les champs
            document.getElementById('current-password').value = '';
            document.getElementById('new-password').value = '';
            document.getElementById('confirm-password').value = '';
        }, 1500);
    } catch (error) {
        messageEl.innerHTML = `<span style="color:#ff4444;">❌ Erreur: ${error.message}</span>`;
    }
};

        function initRadar() {
            const canvas = document.getElementById('radar-canvas');
            if (!canvas) return;
            
            const ctx = canvas.getContext('2d');
            let angle = 0;
            
            function resizeCanvas() {
                const container = canvas.parentElement;
                if (container) {
                    canvas.width = container.clientWidth;
                    canvas.height = container.clientHeight;
                }
            }
            
            function draw() {
                resizeCanvas();
                ctx.clearRect(0, 0, canvas.width, canvas.height);
                
                ctx.strokeStyle = 'rgba(0,255,0,0.2)';
                for (let i = 1; i <= 4; i++) {
                    ctx.beginPath();
                    ctx.arc(canvas.width/2, canvas.height/2, i * Math.min(canvas.width, canvas.height)/8, 0, 2*Math.PI);
                    ctx.stroke();
                }
                
                for (let i = 0; i < 8; i++) {
                    ctx.beginPath();
                    ctx.moveTo(canvas.width/2, canvas.height/2);
                    ctx.lineTo(canvas.width/2 + Math.cos(i*Math.PI/4)*canvas.width/2, 
                               canvas.height/2 + Math.sin(i*Math.PI/4)*canvas.height/2);
                    ctx.stroke();
                }
                
                ctx.strokeStyle = '#0f0';
                ctx.lineWidth = 2;
                ctx.beginPath();
                ctx.moveTo(canvas.width/2, canvas.height/2);
                ctx.lineTo(canvas.width/2 + Math.cos(angle)*canvas.width/2, 
                           canvas.height/2 + Math.sin(angle)*canvas.height/2);
                ctx.stroke();
                
                angle += 0.02;
                requestAnimationFrame(draw);
            }
            
            draw();
        }

        async function loadTrends() {

            const interval = document.getElementById('analytics-interval')?.value;

                    if (!ctx) return;

                        type: 'line',
                        data: {
                            labels: [],

                                label: `${sensor}`,
                                data: [],
                                borderColor: '#00ff00',
                                backgroundColor: 'rgba(0, 255, 0, 0.1)',
                                tension: 0.4
                            }]
                        },
                        options: {
                            responsive: true,
                            maintainAspectRatio: false,
                            scales: {
                                x: {
                                    type: 'time',

                            plugins: {
                                legend: { labels: { color: '#00ff00' } }
                            }
                        }
                    });
                }
                
                if (trendsChart) {
                    trendsChart.data.labels = data.map(d => d.bucket);
                    trendsChart.data.datasets = [{
                        label: `${sensor} - Moyenne`,
                        data: data.map(d => ({ x: d.bucket, y: d.avg_value })),
                        borderColor: '#00ff00',
                        backgroundColor: 'rgba(0, 255, 0, 0.1)',
                        tension: 0.4

                        });

                            label: 'Max',

                            borderColor: '#ffff00',
                            backgroundColor: 'transparent',
                            borderDash: [5, 5],
                            pointRadius: 0
                        });
                    }
                    
                    trendsChart.update();
                }
                
                const stats = document.getElementById('analytics-stats');
                if (stats && data.length > 0) {
                    const last = data[0];
                    stats.innerHTML = `

                        <div class="info-row">Min: <span class="info-value">${last.min_value.toFixed(2)}</span></div>
                        <div class="info-row">Max: <span class="info-value">${last.max_value.toFixed(2)}</span></div>
                        <div class="info-row">Écart-type: <span class="info-value">${last.deviation?.toFixed(2) || 'N/A'}</span></div>

            if (!unitId) return;

                const data = await api.getTrends(unitId, { 
                    interval: '1 hour', 
                    range: '1 week', 
                    sensorType 
                });
                
                const ctx = document.getElementById('heatmap-chart')?.getContext('2d');

                if (heatmapChart) heatmapChart.destroy();
                
                const days = ['Dim', 'Lun', 'Mar', 'Mer', 'Jeu', 'Ven', 'Sam'];
                const heatData = [];
                
                data.forEach(d => {
                    const date = new Date(d.bucket);
                    const day = date.getDay();
                    const hour = date.getHours();
                    heatData.push({
                        x: hour,
                        y: day,
                        v: d.avg_value
                    });
                });
                
                heatmapChart = new Chart(ctx, {
                    type: 'scatter',
                    data: {
                        datasets: [{
                            label: sensorType,
                            data: heatData,
                            backgroundColor: (ctx) => {
                                const value = ctx.raw.v;
                                if (!value) return '#000000';
                                const min = 35;
                                const max = 39;
                                const intensity = Math.min(1, Math.max(0, (value - min) / (max - min)));
                                return `rgba(255, ${255 - intensity*200}, 0, 0.8)`;
                            },
                            pointRadius: 15,
                            pointHoverRadius: 18
                        }]
                    },
                    options: {
                        responsive: true,
                        maintainAspectRatio: false,

                            x: {
                                type: 'linear',
                                position: 'top',
                                min: -0.5,
                                max: 23.5,
                                ticks: {
                                    stepSize: 3,
                                    color: '#00ff00',
                                    callback: (v) => `${v}h`
                                },
                                grid: { color: 'rgba(0,255,0,0.1)' }
                            },
                            y: {
                                type: 'linear',

    class CircuitBreaker {
        constructor() {
            this.failures = 0;
            this.state = 'CLOSED';
            this.nextAttempt = 0;
            this.threshold = 3;
            this.resetTimeout = 60000;
            this.baseDelay = 1000;
        }
        
        async execute(fn, url) {
            if (this.state === 'OPEN') {
                if (Date.now() > this.nextAttempt) {
                    console.log(`🔄 Circuit breaker en test (HALF_OPEN)`);
                    this.state = 'HALF_OPEN';
                } else {
                    console.log(`🔌 Circuit breaker OPEN - réessai dans ${Math.ceil((this.nextAttempt - Date.now())/1000)}s`);
                    throw new Error('Service temporairement indisponible');
                }
            }
            
            try {
                const result = await fn();
                if (this.state !== 'CLOSED') {
                    console.log(`✅ Circuit breaker refermé`);
                }
                this.failures = 0;
                this.state = 'CLOSED';
                return result;
            } catch (error) {
                this.failures++;
                console.log(`❌ Échec (${this.failures}/${this.threshold}) pour ${url}`);
                
                if (this.failures >= this.threshold && this.state !== 'OPEN') {
                    this.state = 'OPEN';
                    const delay = Math.min(this.baseDelay * Math.pow(2, this.failures), 30000);
                    this.nextAttempt = Date.now() + delay;
                    console.log(`🔌 Circuit breaker OPEN - pause de ${delay/1000}s`);
                }
                throw error;
            }
        }
    }
    
    // Créer une instance par endpoint
    const circuitBreakers = {
        vests: new CircuitBreaker(),
        positions: new CircuitBreaker(),
        alerts: new CircuitBreaker(),
        dashboard: new CircuitBreaker(),
        summary: new CircuitBreaker()
    };
    
    // Fonction fetch avec timeout
    async function fetchWithTimeout(url, options = {}, timeout = 8000) {
        const controller = new AbortController();
        const timeoutId = setTimeout(() => controller.abort(), timeout);
        
        try {
            const response = await fetch(url, {
                ...options,
                signal: controller.signal
            });
            clearTimeout(timeoutId);
            return response;
        } catch (error) {
            clearTimeout(timeoutId);
            if (error.name === 'AbortError') {
                throw new Error(`Timeout (${timeout}ms) pour ${url}`);
            }
            throw error;
        }
    }
    
    // Remplacer les méthodes API originales par les versions avec Circuit Breaker

// =====================================================
// CACHE INTELLIGENT POUR DASHBOARD ULTRA-RAPIDE
// =====================================================
// - Données instantanées (même périmées)
// - Rafraîchissement en arrière-plan
// - TimescaleDB = source de vérité
// =====================================================

(function() {
    'use strict';
    
    console.log('🚀 Initialisation du cache intelligent...');
    
    // ===== 1. CONFIGURATION =====
    const CACHE_CONFIG = {
        // Type de donnée : { ttl: temps avant rafraîchissement, important: booléen }
        vests: { ttl: 30000, important: true },        // 30 secondes
        positions: { ttl: 5000, important: true },      // 5 secondes
        alerts: { ttl: 2000, important: true },         // 2 secondes
        dashboard: { ttl: 10000, important: true },     // 10 secondes
        analytics: { ttl: 300000, important: false }    // 5 minutes (peu important)
    };
    
    // ===== 2. CACHE PRINCIPAL =====
    window.fastCache = {
        // Stockage
        _store: new Map(),
        _listeners: new Map(),
        _pendingRefreshes: new Set(),
        
        // ===== 2.1 Mettre en cache =====
        set(key, value, type = 'vests') {
            this._store.set(key, {
                value: value,
                timestamp: Date.now(),
                type: type,
                hits: (this._store.get(key)?.hits || 0) + 1
            });
            
            console.log(`💾 Cache mis à jour: ${key} (${type})`);
            
            // Notifier les listeners que des données fraîches sont arrivées
            this._notifyListeners(key, value);
            
            return value;
        },
        
        // ===== 2.2 Récupérer du cache (TOUJOURS instantané) =====
        get(key, type = 'vests') {
            const cached = this._store.get(key);
            
            // Pas de cache du tout
            if (!cached) {
                console.log(`❌ Cache MISS: ${key}`);
                return null;
            }
            
            // Calculer l'âge
            const age = Date.now() - cached.timestamp;
            const config = CACHE_CONFIG[type] || CACHE_CONFIG.vests;
            
            // Même si expiré, on retourne la donnée (mais avec indicateur)
            if (age > config.ttl) {
                console.log(`⚠️ Cache STALE: ${key} (${age}ms > ${config.ttl}ms) - mais utilisable`);
                return {
                    ...cached.value,
                    _stale: true,
                    _age: age,
                    _key: key,
                    _type: type
                };
            }
            
            // Cache valide
            console.log(`✅ Cache HIT: ${key} (${age}ms)`);
            return cached.value;
        },
        
        // ===== 2.3 Récupération intelligente =====
        async getOrFetch(key, type, fetchFn, options = {}) {
            const cached = this.get(key, type);
            
            // Si on a des données en cache (même périmées), on les retourne IMMÉDIATEMENT
            if (cached) {
                // Déclencher un rafraîchissement en arrière-plan si nécessaire
                this._refreshIfNeeded(key, type, fetchFn);
                return cached;
            }
            
            // Pas de cache du tout : on attend TimescaleDB (premier chargement)
            console.log(`🔄 Premier chargement ${key} depuis TimescaleDB...`);
            try {
                const fresh = await fetchFn();
                this.set(key, fresh, type);
                return fresh;
            } catch (error) {
                console.error(`❌ TimescaleDB error pour ${key}:`, error);
                throw error;
            }
        },
        
        // ===== 2.4 Rafraîchissement en arrière-plan =====
        async _refreshIfNeeded(key, type, fetchFn) {
            const cached = this._store.get(key);
            if (!cached) return;
            
            const age = Date.now() - cached.timestamp;
            const config = CACHE_CONFIG[type] || CACHE_CONFIG.vests;
            
            // Si le cache est encore valide, pas besoin de rafraîchir
            if (age < config.ttl) return;
            
            // Éviter les rafraîchissements multiples simultanés
            if (this._pendingRefreshes.has(key)) return;
            
            this._pendingRefreshes.add(key);
            
            console.log(`🔄 Rafraîchissement async ${key}...`);
            
            try {
                const fresh = await fetchFn();
                this.set(key, fresh, type);
                console.log(`✅ Rafraîchissement réussi: ${key}`);
            } catch (error) {
                console.log(`⚠️ Échec rafraîchissement ${key} (on garde l'ancien)`);
            } finally {
                this._pendingRefreshes.delete(key);
            }
        },
        
        // ===== 2.5 Système de listeners =====
        subscribe(key, callback) {
            if (!this._listeners.has(key)) {
                this._listeners.set(key, []);
            }
            this._listeners.get(key).push(callback);
            
            // Retourner fonction pour se désabonner
            return () => {
                const callbacks = this._listeners.get(key);
                if (callbacks) {
                    const index = callbacks.indexOf(callback);
                    if (index > -1) callbacks.splice(index, 1);
                }
            };
        },
        
        _notifyListeners(key, data) {
            const callbacks = this._listeners.get(key) || [];
            callbacks.forEach(cb => {
                try {
                    cb(data);
                } catch (e) {
                    console.error(`❌ Erreur listener ${key}:`, e);
                }
            });
        },
        
        // ===== 2.6 Utilitaires =====
        invalidate(key) {
            this._store.delete(key);
            console.log(`🗑️ Cache invalidé: ${key}`);
        },
        
        clear() {
            this._store.clear();
            console.log('🧹 Cache complètement vidé');
        },
        
        stats() {
            const stats = {};
            this._store.forEach((value, key) => {
                stats[key] = {
                    age: Date.now() - value.timestamp,
                    type: value.type,
                    hits: value.hits
                };
            });
            return stats;
        }
    };

(function() {
    'use strict';
    
    console.log('🔍 Initialisation du système de recherche intelligent...');
    
    // ===== 1. CONFIGURATION =====
    const SEARCH_CONFIG = {
        minChars: 2,           // Minimum de caractères
        localDebounce: 0,       // Filtrage local = instantané
        apiDebounce: 400,       // Appels API = 400ms

    // ===== 2. FONCTION DEBOUNCE GÉNÉRIQUE =====
    function debounce(func, delay) {
        let timeoutId;
        return function(...args) {
            clearTimeout(timeoutId);
            timeoutId = setTimeout(() => func.apply(this, args), delay);
        };
    }
    
    // ===== 3. GESTIONNAIRE DE RECHERCHE PRINCIPAL =====
    class SearchManager {
        constructor(inputId, options = {}) {
    this.input = document.getElementById(inputId);
    if (!this.input) return null;
    
    this.inputId = inputId;  // ← Déjà présent normalement
    this.inputElement = this.input; // ← AJOUTEZ AUSSI CETTE LIGNE
    this.options = {
        type: options.type || 'local',
        minChars: options.minChars || SEARCH_CONFIG.minChars,
        onSearch: options.onSearch || (() => {}),
        onError: options.onError || (() => {}),
        ...options
    };
    
    this.lastTerm = '';
    this.searchCount = 0;
    this.init();
}
        
        init() {
            // Créer l'indicateur de recherche
            this.createSearchHint();
            
            // Créer le gestionnaire selon le type
            if (this.options.type === 'api') {
                this.handler = debounce(
                    (term) => this.performSearch(term),
                    SEARCH_CONFIG.apiDebounce
                );
            } else {
                // Local = instantané
                this.handler = (term) => this.performSearch(term);
            }
            
            // Attacher l'événement
            this.input.addEventListener('input', (e) => {
                const term = e.target.value.trim();
                this.handleInput(term);
            });
            
            // Raccourci clavier pour effacer
            this.input.addEventListener('keydown', (e) => {
                if (e.key === 'Escape') {
                    this.input.value = '';
                    this.handleInput('');
                }
            });
            
           console.log(`✅ Recherche ${this.options.type} initialisée: ${this.inputId || this.input?.id || 'inconnu'}`);
        }

        handleInput(term) {
            // Vérifier la longueur minimale
            if (term.length < this.options.minChars && term.length > 0) {
                this.showHint(`Minimum ${this.options.minChars} caractères`);
                return;
            }
            
            // Mettre à jour le compteur
            this.searchCount++;
            
            // Afficher un indicateur visuel
            this.showSearchActivity(term);
            
            // Appeler le gestionnaire approprié
            this.handler(term);
        }
        
        async performSearch(term) {
            // Éviter les recherches identiques
            if (term === this.lastTerm && term.length > 0) return;
            this.lastTerm = term;
            
            try {
                // Déterminer le type de recherche
                const searchType = this.determineSearchType(term);
                
                // Log pour debug
                console.log(`🔍 [${searchType}] Recherche: "${term}"`);
                
                // Exécuter la recherche
                const results = await this.options.onSearch(term, searchType);
                
                // Mettre à jour l'UI
                this.updateResultsCount(results);
                
            } catch (error) {
                console.error('❌ Erreur recherche:', error);
                this.options.onError(error);
            }
        }
        
        determineSearchType(term) {
            // Si c'est un ID exact (format VEST-XXX)
            if (/^VEST-\d{3}$/i.test(term)) {
                return 'exact';
            }
            
            // Si c'est très court
            if (term.length <= 2) {
                return 'prefix';
            }
            
            // Si on a des données en cache
            if (window.fastCache?.get('all_vests')) {
                return 'local';
            }
            
            return 'api';
        }
        
        createSearchHint() {
            if (!SEARCH_CONFIG.showHints) return;
            
            this.hintElement = document.createElement('div');
            this.hintElement.className = 'search-hint';
            this.hintElement.style.cssText = `

            const container = this.input.parentElement;
            if (container) {
                container.style.position = 'relative';
                container.appendChild(this.hintElement);
            }
        }
        
        showHint(message) {
            if (!this.hintElement) return;
            
            this.hintElement.textContent = message;
            this.hintElement.style.opacity = '1';
            
            setTimeout(() => {
                this.hintElement.style.opacity = '0';
            }, 1500);
        }
        
        showSearchActivity(term) {
            if (!this.hintElement) return;
            
            if (term.length === 0) {
                this.hintElement.textContent = '✨';
            } else if (term.length < this.options.minChars) {
                this.hintElement.textContent = `⚠️ ${this.options.minChars - term.length}`;
            } else {
                this.hintElement.textContent = '🔍';
            }
            this.hintElement.style.opacity = '0.8';
        }
        
        updateResultsCount(results) {
            if (!this.hintElement) return;
            
            const count = Array.isArray(results) ? results.length : 0;
            if (count > 0) {
                this.hintElement.textContent = `${count} résultat${count > 1 ? 's' : ''}`;
                this.hintElement.style.color = '#00ff00';

                this.hintElement.style.color = '#ff8800';
            }
            
            setTimeout(() => {
                this.hintElement.style.opacity = '0';
            }, 2000);
        }
    }

function setupIntelligentSearches() {
    
    // Vérifier que les éléments existent avant de créer les SearchManager
    const setupSearch = (inputId, options) => {
        const element = document.getElementById(inputId);
        if (element) {
            return new SearchManager(inputId, options);
        } else {
            console.log(`⚠️ Élément #${inputId} non trouvé, recherche non activée`);
            return null;
        }
    };
    
    // Dashboard search
    setupSearch('dashboard-search', {
        type: 'hybrid',
        minChars: 2,
        onSearch: async (term, searchType) => {
            if (searchType === 'local' || searchType === 'prefix') {
                const cached = window.fastCache?.get('all_vests');
                if (cached) {
                    const results = cached.filter(u => 
                        u.id.toLowerCase().includes(term) ||
                        (u.soldier_name && u.soldier_name.toLowerCase().includes(term))
                    );
                    
                    if (window.updateUnitsDisplay) {
                        window.updateUnitsDisplay(results);
                    }
                    
                    return results;
                }
            }
            
            if (window.api && searchType === 'api') {
                try {
                    const vests = await window.api.getVests();
                    const results = vests.filter(u => 
                        u.id.toLowerCase().includes(term) ||
                        (u.soldier_name && u.soldier_name.toLowerCase().includes(term))
                    );
                    
                    if (window.updateUnitsDisplay) {
                        window.updateUnitsDisplay(results);
                    }
                    
                    return results;
                } catch (error) {
                    console.error('❌ Erreur API recherche:', error);
                    return [];
                }
            }
            
            return [];
        }
    });
    
    // Units search
    setupSearch('units-search', {
        type: 'local',
        minChars: 1,
        onSearch: (term) => {
            const tableBody = document.getElementById('units-table-body');
            if (!tableBody) return [];
            
            const rows = tableBody.getElementsByTagName('tr');
            let visibleCount = 0;
            
            for (let row of rows) {
                const text = row.textContent.toLowerCase();
                const visible = text.includes(term.toLowerCase());
                row.style.display = visible ? '' : 'none';
                if (visible) visibleCount++;
            }
            
            return { count: visibleCount };
        }
    });
    
    // Sensors search
    setupSearch('sensors-filter', {
        type: 'local',
        minChars: 1,
        onSearch: (term) => {
            const tableBody = document.getElementById('sensors-table-body');
            if (!tableBody) return [];
            
            const rows = tableBody.getElementsByTagName('tr');
            for (let row of rows) {
                const text = row.textContent.toLowerCase();
                row.style.display = text.includes(term.toLowerCase()) ? '' : 'none';
            }
        }
    });
}
    
    // ===== 5. RECHERCHE RAPIDE D'ID (sans debounce) =====
    window.quickIdSearch = function(id) {
        if (!id || id.length < 7) return; // VEST-001 = 8 chars
        
        console.log(`⚡ Recherche rapide ID: ${id}`);
        
        // Chercher dans le cache d'abord
        const cached = window.fastCache?.get('all_vests');
        if (cached) {
            const unit = cached.find(u => u.id === id);
            if (unit) {
                showErrorMessage(`✅ Unité trouvée: ${unit.soldier_name}`, 'success');
                // Mettre en évidence dans l'interface
                highlightUnit(unit.id);
                return;
            }
        }
        
        // Sinon appel API immédiat
        if (window.api) {
            window.api.getVest(id).then(unit => {
                if (unit) {
                    showErrorMessage(`✅ Unité trouvée: ${unit.soldier_name}`, 'success');
                    highlightUnit(unit.id);
                } else {
                    showErrorMessage(`❌ Unité ${id} non trouvée`, 'error');
                }
            }).catch(() => {
                showErrorMessage(`❌ Erreur recherche ${id}`, 'error');
            });
        }
    };
    
    function highlightUnit(unitId) {
        // Mettre en évidence l'unité dans le tableau
        const rows = document.querySelectorAll('#units-table-body tr');
        rows.forEach(row => {
            if (row.textContent.includes(unitId)) {
                row.style.background = 'rgba(0, 255, 0, 0.3)';
                row.style.transition = 'background 1s';
                setTimeout(() => {
                    row.style.background = '';
                }, 2000);
            }
        });
    }

    document.addEventListener('keydown', (e) => {
        // Ctrl+F : focus recherche et suggère format ID
        if (e.ctrlKey && e.key === 'f') {
            e.preventDefault();
            const search = document.getElementById('dashboard-search');
            if (search) {
                search.focus();
                search.select();
                showErrorMessage('🔍 Tapez un ID (ex: VEST-001) pour recherche rapide', 'info');
            }
        }
        
        // Ctrl+G : aller à une unité spécifique
        if (e.ctrlKey && e.key === 'g') {
            e.preventDefault();
            const id = prompt("ID de l'unité (ex: VEST-001):");
            if (id) {
                window.quickIdSearch(id.toUpperCase());
            }
        }
        
        // Ctrl+Shift+F : recherche avancée (API)
        if (e.ctrlKey && e.shiftKey && e.key === 'F') {
            e.preventDefault();
            showErrorMessage('🌐 Mode recherche avancée activé', 'info');
            // Forcer le mode API pour la prochaine recherche
            window.forceApiSearch = true;
        }
    });
    
    // ===== 7. INITIALISATION =====
    setTimeout(() => {
        setupIntelligentSearches();
        console.log('✅ Système de recherche intelligent activé');
        console.log('📌 Raccourcis:');
        console.log('   • Ctrl+F : Focus recherche');
        console.log('   • Ctrl+G : Recherche par ID');
        console.log('   • Esc : Effacer recherche');
        console.log('   • Ctrl+Shift+F : Mode API forcé');
    }, 1000);
    
    // Exposer les classes
    window.SearchManager = SearchManager;
    
})();

(function() {

        // Exporter les unités
        exporterUnites: function(format = 'json') {
            const unites = window.fastCache?.get('all_vests') || [];
            
            if (!unites.length) {
                alert('Aucune unité à exporter');
                return;
            }
            
            const data = {
                type: 'unités',
                date: new Date().toISOString(),
                total: unites.length,
                unites: unites.map(u => ({
                    id: u.id,
                    nom: u.soldier_name || 'N/A',
                    statut: u.status || 'inconnu',
                    batterie: u.battery_level || 0,
                    temperature: u.last_temp || 'N/A',
                    rythme_cardiaque: u.last_heart_rate || 'N/A',
                    position: u.latitude && u.longitude ? 
                        `${u.latitude}, ${u.longitude}` : 'N/A',
                    derniere_maj: u.last_seen || new Date().toISOString()
                }))
            };
            
            this.telecharger(data, `unites_${this.formaterDate()}`, format);
        },
        
        // Exporter les missions
        exporterMissions: function(format = 'json') {
            // Récupérer les missions depuis le stockage local ou l'état global
            const missions = window.missionsData || this.getMissionsParDefaut();
            
            const data = {
                type: 'missions',
                date: new Date().toISOString(),
                total: missions.length,
                missions: missions.map(m => ({
                    nom: m.nom || 'Sans nom',
                    type: m.type || 'Non défini',
                    priorite: m.priorite || 'moyenne',
                    date_debut: m.date_debut || new Date().toISOString(),
                    description: m.description || '',
                    objectifs: m.objectifs || [],
                    unites_affectees: m.unites || [],
                    statut: m.statut || 'brouillon'
                }))
            };
            
            this.telecharger(data, `missions_${this.formaterDate()}`, format);
        },
        
        // Télécharger le fichier
        telecharger: function(data, nomFichier, format) {
            let contenu;
            let typeMime;
            
            if (format === 'json') {
                contenu = JSON.stringify(data, null, 2);
                typeMime = 'application/json';
                nomFichier += '.json';
            } else {
                // Format CSV
                contenu = this.convertirEnCSV(data);
                typeMime = 'text/csv';
                nomFichier += '.csv';
            }
            
            const blob = new Blob([contenu], { type: typeMime });
            const url = URL.createObjectURL(blob);
            const lien = document.createElement('a');
            lien.href = url;
            lien.download = nomFichier;
            document.body.appendChild(lien);
            lien.click();
            document.body.removeChild(lien);
            URL.revokeObjectURL(url);
            
            console.log(`✅ Exporté: ${nomFichier}`);
        },
        
        // Convertir en CSV
        convertirEnCSV: function(data) {
            if (data.type === 'unités') {
                return this.unitesVersCSV(data);
            } else if (data.type === 'missions') {
                return this.missionsVersCSV(data);
            }
            return '';
        },
        
        unitesVersCSV: function(data) {
            let csv = 'ID,Nom,Statut,Batterie,Température,HR,Position,Dernière MAJ\n';
            
            data.unites.forEach(u => {
                csv += `"${u.id}","${u.nom}","${u.statut}",${u.batterie},${u.temperature},${u.rythme_cardiaque},"${u.position}","${u.derniere_maj}"\n`;
            });
            
            return csv;
        },
        
        missionsVersCSV: function(data) {
            let csv = 'Nom,Type,Priorité,Date,Description,Objectifs,Unités,Statut\n';
            
            data.missions.forEach(m => {
                csv += `"${m.nom}","${m.type}","${m.priorite}","${m.date_debut}","${m.description}","${m.objectifs.join('|')}","${m.unites_affectees.join('|')}","${m.statut}"\n`;
            });
            
            return csv;
        },
        
        // Missions par défaut
        getMissionsParDefaut: function() {

                    statut: 'planification'

        },
        
        // Formater la date pour le nom de fichier
        formaterDate: function() {
            const d = new Date();
            return `${d.getFullYear()}-${String(d.getMonth()+1).padStart(2,'0')}-${String(d.getDate()).padStart(2,'0')}`;
        }

        if (e.key === 'F11') {
            e.preventDefault(); // Empêcher le F11 natif

        }
    });

    function creerGraphiqueTempsReel(canvasId, title, unit, color) {
    const canvas = document.getElementById(canvasId);
    if (!canvas) return null;
    
    const ctx = canvas.getContext('2d');
    
    // Version simplifiée sans time scale
    const chart = new Chart(ctx, {
        type: 'line',
        data: {
            labels: Array(20).fill('').map((_, i) => `${i*5}s`),
            datasets: [{
                label: title,
                data: Array(20).fill(0).map(() => Math.random() * 100),
                borderColor: color,
                backgroundColor: color + '20',
                tension: 0.4

    async function loadMessages() {
        try {
            const container = document.getElementById('chat-messages-container');
            if (!container) return;
            
            container.innerHTML = '<div class="chat-message system" style="align-self: center;">Chargement des messages...</div>';
            
            let query = supabase
                .from('chat_messages')
                .select('*')
                .order('created_at', { ascending: true })
                .limit(100);
            
            // Filtrer par canal
            if (currentChannel !== 'all') {
                query = query.eq('channel', currentChannel);
            }
            
            // Si destinataire spécifique
            if (currentRecipient !== 'all') {
                query = query.contains('recipients', [currentRecipient]);
            }
            
            const { data, error } = await query;
            
            if (error) throw error;
            
            messages = data || [];
            console.log(`📨 ${messages.length} messages chargés pour ${currentChannel}`);
            displayMessages();
            
        } catch (error) {
            console.error('❌ Erreur chargement messages:', error);
            showError('Impossible de charger l\'historique');
            
            // Afficher un message d'erreur dans le container
            const container = document.getElementById('chat-messages-container');

    // ===== AFFICHER LES MESSAGES =====
    function displayMessages() {
        const container = document.getElementById('chat-messages-container');
        if (!container) return;
        
        if (messages.length === 0) {
            container.innerHTML = `
                <div class="chat-message system" style="align-self: center;">
                    <i class="fas fa-shield-alt"></i> 
                    Aucun message dans ce canal. Commencez la conversation !
                </div>
            `;
            return;
        }
        
        // Filtrer par priorité si nécessaire
        let filteredMessages = messages;
        if (hideNormalMessages) {
            filteredMessages = messages.filter(m => m.priority !== 'normal');
        }
        
        let html = '';
        filteredMessages.forEach(msg => {
            html += renderMessage(msg);
        });
        
        container.innerHTML = html;
        
        // Scroll en bas
        container.scrollTop = container.scrollHeight;
    }
    
    // ===== RENDU D'UN MESSAGE =====
    function renderMessage(msg) {
        const isOwn = msg.sender_id === currentUser?.id;
        const time = new Date(msg.created_at).toLocaleTimeString('fr-FR', {
            hour: '2-digit',
            minute: '2-digit'
        });
        
        let priorityClass = '';
        let priorityIcon = '💬';
        if (msg.priority === 'urgent') {
            priorityClass = 'urgent';
            priorityIcon = '🔴';
        } else if (msg.priority === 'important') {
            priorityClass = 'important';
            priorityIcon = '⚠️';
        }
        
        let senderClass = isOwn ? 'own' : 'other';
        if (msg.message_type === 'system') senderClass = 'system';
        
        // Afficher les destinataires si ce n'est pas "all"
        let recipientsInfo = '';
        if (msg.recipients && !msg.recipients.includes('all') && msg.recipients.length > 0) {
            recipientsInfo = `<div style="font-size: 9px; color: #888; margin-top: 2px;">📨 À: ${msg.recipients.join(', ')}</div>`;
        }
        
        // Vérifier si le message a des coordonnées
        const hasCoordinates = msg.has_coordinates && msg.latitude && msg.longitude;
        
        return `
            <div class="chat-message ${senderClass} ${priorityClass}" data-id="${msg.id}">
                <div class="message-header">
                    <span class="message-sender ${msg.sender_name === 'COMMANDANT' ? 'commandant' : ''}">
                        ${priorityIcon} ${msg.sender_name}
                        ${msg.sender_role ? `<span style="color:#888; font-size:10px; margin-left:5px;">(${msg.sender_role})</span>` : ''}
                    </span>
                    <span class="message-time">${time}</span>
                </div>
                <div class="message-content">
                    ${msg.message}
                    ${hasCoordinates ? `
                        <div style="margin-top: 8px; padding: 5px; background: rgba(0,255,0,0.1); border-radius: 4px; font-size: 11px;">
                            <i class="fas fa-map-marker-alt" style="color: #ff0;"></i> 
                            ${msg.latitude.toFixed(4)}, ${msg.longitude.toFixed(4)}
                        </div>
                    ` : ''}
                    ${recipientsInfo}
                </div>
                <div class="message-footer">
                    ${msg.priority === 'urgent' ? '<span style="color:#ff4444;">🔴 URGENT</span>' : ''}
                    ${msg.priority === 'important' ? '<span style="color:#ff8800;">⚠️ IMPORTANT</span>' : ''}
                </div>
            </div>
        `;
    }
    
    // ===== SOUSCRIRE AUX NOUVEAUX MESSAGES =====
    function subscribeToMessages() {
        if (!supabase) return;
        
        supabase
            .channel('chat-room')
            .on('postgres_changes', 
                { event: 'INSERT', schema: 'public', table: 'chat_messages' },
                (payload) => {
                    const newMsg = payload.new;
                    
                    // Vérifier si le message concerne le canal actuel
                    let shouldDisplay = false;
                    
                    if (newMsg.channel === currentChannel) {
                        shouldDisplay = true;
                    }
                    
                    // Vérifier les destinataires
                    if (newMsg.recipients && !newMsg.recipients.includes('all')) {
                        if (newMsg.recipients.includes(currentUser.id) || 
                            newMsg.recipients.includes(currentUser.name)) {
                            shouldDisplay = true;
                        }
                    }
                    
                    if (shouldDisplay) {
                        messages.push(newMsg);
                        
                        if (messages.length > 200) {
                            messages.shift();
                        }
                        
                        displayMessages();
                        
                        // Notification sonore
                        if (soundEnabled && newMsg.sender_id !== currentUser?.id) {
                            playNotificationSound(newMsg.priority);
                        }
                    }
                    
                    // Mettre à jour les badges des canaux
                    updateChannelBadges();
                }
            )
            .subscribe();
    }
    
    // ===== ENVOYER UN MESSAGE =====
    async function sendMessage() {
        const input = document.getElementById('chat-input');
        if (!input) return;
        
        const message = input.value.trim();
        if (!message) return;
        
        // Déterminer les destinataires
        let recipients = ['all'];
        if (currentRecipient !== 'all') {
            recipients = [currentRecipient];
        }
        
        // Vérifier partage de position
        const sharePosition = document.getElementById('chat-share-position')?.checked;
        let latitude = null, longitude = null;
        
        if (sharePosition && navigator.geolocation) {
            try {
                const position = await new Promise((resolve, reject) => {
                    navigator.geolocation.getCurrentPosition(resolve, reject, {
                        timeout: 5000,
                        maximumAge: 60000
                    });
                });
                latitude = position.coords.latitude;
                longitude = position.coords.longitude;
            } catch (error) {
                console.log('⚠️ Position non disponible');
            }
        }
        
        try {
            // Vérifier que la table existe
            const { error: testError } = await supabase
                .from('chat_messages')
                .select('count')
                .limit(1);
                
            if (testError && testError.code === '42P01') {
                showError('La table chat_messages n\'existe pas dans Supabase');
                return;
            }
            
            const { error } = await supabase
                .from('chat_messages')
                .insert([{
                    sender_id: currentUser.id,
                    sender_name: currentUser.name,
                    sender_role: currentUser.role,
                    message: message,
                    channel: currentChannel,
                    priority: currentPriority,
                    message_type: 'text',
                    has_coordinates: !!(latitude && longitude),
                    latitude: latitude,
                    longitude: longitude,
                    recipients: recipients
                }]);
            
            if (error) throw error;
            
            input.value = '';
            
            // Son d'envoi
            if (soundEnabled) playSendSound();
            
        } catch (error) {
            console.error('❌ Erreur envoi:', error);
            showError('Erreur lors de l\'envoi: ' + (error.message || 'inconnue'));
        }
    }

    // ===== NOTIFICATIONS SONORES =====
    function playNotificationSound(priority) {
        const audioCtx = window.audioCtx;
        if (!audioCtx || audioCtx.state === 'closed') return;
        
        if (audioCtx.state === 'suspended') {
            audioCtx.resume();
            return;
        }
        
        try {
            const osc = audioCtx.createOscillator();
            const gain = audioCtx.createGain();
            
            osc.connect(gain);
            gain.connect(audioCtx.destination);
            
            gain.gain.setValueAtTime(0.1, audioCtx.currentTime);
            
            if (priority === 'urgent') {
                osc.frequency.setValueAtTime(880, audioCtx.currentTime);
                osc.frequency.setValueAtTime(440, audioCtx.currentTime + 0.2);
                gain.gain.exponentialRampToValueAtTime(0.01, audioCtx.currentTime + 0.5);
                osc.start();
                osc.stop(audioCtx.currentTime + 0.5);
            } else if (priority === 'important') {
                osc.frequency.setValueAtTime(660, audioCtx.currentTime);
                gain.gain.exponentialRampToValueAtTime(0.01, audioCtx.currentTime + 0.3);
                osc.start();
                osc.stop(audioCtx.currentTime + 0.3);
            } else {
                osc.frequency.setValueAtTime(440, audioCtx.currentTime);
                gain.gain.exponentialRampToValueAtTime(0.01, audioCtx.currentTime + 0.2);
                osc.start();
                osc.stop(audioCtx.currentTime + 0.2);
            }
        } catch (e) {
            console.log('⚠️ Erreur audio:', e);
        }
    }
    
    function playSendSound() {
        const audioCtx = window.audioCtx;
        if (!audioCtx) return;
        
        try {
            const osc = audioCtx.createOscillator();
            const gain = audioCtx.createGain();
            
            osc.connect(gain);
            gain.connect(audioCtx.destination);
            
            gain.gain.setValueAtTime(0.05, audioCtx.currentTime);
            osc.frequency.setValueAtTime(523.25, audioCtx.currentTime);
            osc.frequency.setValueAtTime(659.25, audioCtx.currentTime + 0.1);
            gain.gain.exponentialRampToValueAtTime(0.01, audioCtx.currentTime + 0.2);
            
            osc.start();
            osc.stop(audioCtx.currentTime + 0.2);
        } catch (e) {
            console.log('⚠️ Erreur audio:', e);
        }
    }

                    if (chatTab?.classList.contains('active')) {
                        initChatTab();
                        checkChatTable();
                    }
                }
            });

        const chatTab = document.getElementById('chat-tab');
        if (chatTab) {
            observer.observe(chatTab, { attributes: true });
        }
        
        // Si l'onglet est déjà actif

            initChatTab();

    // Supprimer l'ancien message
    const oldMsg = document.querySelector('.system-status');
    if (oldMsg) oldMsg.remove();
    
    const msgDiv = document.createElement('div');
    msgDiv.className = `system-status ${type}`;
    msgDiv.innerHTML = `
        <i class="fas ${type === 'warning' ? 'fa-exclamation-triangle' : 'fa-info-circle'}"></i>
        ${message}
        <span style="float:right; margin-left:15px; cursor:pointer;" onclick="this.parentElement.remove()">✕</span>
    `;
    document.body.appendChild(msgDiv);
    
    // Auto-fermeture après 10 secondes
    setTimeout(() => {
        if (msgDiv.parentElement) msgDiv.remove();
    }, 10000);
}

// Remplacer les alertes popup par des messages système
function showErrorMessage(message, type = 'error') {
    showSystemMessage(message, type === 'error' ? 'warning' : 'success');
}
