        function playBeep(freq, duration) {
            if (!audioEnabled || !audioCtx || audioCtx.state !== 'running') return;
            try {
                const osc = audioCtx.createOscillator();
                const gain = audioCtx.createGain();
                osc.type = 'square';
                osc.frequency.value = freq;
                gain.gain.value = 0.1;
                osc.connect(gain).connect(audioCtx.destination);
                osc.start();
                osc.stop(audioCtx.currentTime + duration);
            } catch(e) {}

        let radarAngle = 0;
        
        function initRadar() {
            const canvas = document.getElementById('radar-canvas');
            if (!canvas) return;
            
            const ctx = canvas.getContext('2d');
            
            function resize() {
                canvas.width = canvas.parentElement.clientWidth;
                canvas.height = canvas.parentElement.clientHeight;
            }
            
            function draw() {
                resize();
                ctx.clearRect(0, 0, canvas.width, canvas.height);
                
                ctx.strokeStyle = 'rgba(0,255,0,0.2)';
                for (let i = 1; i <= 4; i++) {
                    ctx.beginPath();
                    ctx.arc(canvas.width/2, canvas.height/2, i * Math.min(canvas.width, canvas.height)/8, 0, 2*Math.PI);
                    ctx.stroke();
                }
                
                for (let i = 0; i < 8; i++) {
                    ctx.beginPath();
                    ctx.moveTo(canvas.width/2, canvas.height/2);
                    ctx.lineTo(canvas.width/2 + Math.cos(i*Math.PI/4)*canvas.width/2, 
                               canvas.height/2 + Math.sin(i*Math.PI/4)*canvas.height/2);
                    ctx.strokeStyle = 'rgba(0,255,0,0.2)';
                    ctx.stroke();
                }
                
                ctx.strokeStyle = '#0f0';
                ctx.lineWidth = 2;
                ctx.beginPath();
                ctx.moveTo(canvas.width/2, canvas.height/2);
                ctx.lineTo(canvas.width/2 + Math.cos(radarAngle)*canvas.width/2, 
                           canvas.height/2 + Math.sin(radarAngle)*canvas.height/2);
                ctx.stroke();
                
                radarAngle += 0.02;
                requestAnimationFrame(draw);
            }
            
            draw();
        }

        function setDrawTool(tool) {
            if (!missionMap) return;
            
            if (currentDrawHandler) {
                missionMap.off('click', currentDrawHandler);
            }
            
            switch(tool) {
                case 'select':
                    showAlertPopup('MISSION', 'Mode sélection activé', 'info');
                    break;
                    
                case 'polygon':
                    showAlertPopup('MISSION', 'Dessinez une zone d\'opération', 'info');
                    currentDrawHandler = function(e) {
                        const circle = L.circle(e.latlng, {
                            radius: 500,
                            color: '#ffff00',
                            fillColor: '#ffff00',
                            fillOpacity: 0.2
                        }).addTo(missionMap);
                        missionWaypoints.push(circle);
                    };
                    missionMap.on('click', currentDrawHandler);
                    break;
                    
                case 'marker':
                    showAlertPopup('MISSION', 'Placez un point d\'intérêt', 'info');
                    currentDrawHandler = function(e) {
                        const marker = L.marker(e.latlng, {
                            icon: L.divIcon({
                                className: 'custom-marker',
                                html: '<i class="fas fa-map-pin" style="color:#00ff00; font-size:24px;"></i>',
                                iconSize: [24, 24]
                            })
                        }).addTo(missionMap);
                        marker.bindPopup('Point d\'intérêt').openPopup();
                        missionWaypoints.push(marker);
                    };
                    missionMap.on('click', currentDrawHandler);
                    break;
                    
                case 'rectangle':
                    showAlertPopup('MISSION', 'Dessinez une zone d\'exclusion', 'info');
                    break;
                    
                case 'circle':
                    showAlertPopup('MISSION', 'Dessinez un rayon d\'action', 'info');
                    break;
                    
                case 'line':
                    showAlertPopup('MISSION', 'Dessinez un itinéraire', 'info');
                    break;
                    
                case 'measure':
                    showAlertPopup('MISSION', 'Cliquez pour mesurer une distance', 'info');
                    break;
                    
                case 'clear':
                    clearDrawings();
                    break;
            }
        }

        function addEvent(type, msg, icon, color) {
            eventsData.unshift({
                time: new Date().toLocaleTimeString(),
                type, message: msg, icon, color
            });
            if (eventsData.length > 20) eventsData.pop();
            
            const list = document.getElementById('events-list');
            if (list) {
                list.innerHTML = eventsData.map(e => `
                    <div class="event-item" style="border-left:3px solid ${e.color}">
                        <span class="event-time">${e.time}</span>
                        <span class="event-type"><i class="fas ${e.icon}"></i> ${e.type}</span>
                        <span>${e.message}</span>
                    </div>
                `).join('');
            }
            
            document.getElementById('events-count').textContent = eventsData.length;
        }

        window.changePassword = async function() {
            const current = document.getElementById('current-password')?.value;
            const newPass = document.getElementById('new-password')?.value;
            const confirm = document.getElementById('confirm-password')?.value;
            const msg = document.getElementById('password-message');
            
            if (!current || !newPass || !confirm) {
                msg.innerHTML = '❌ Tous les champs requis';
                msg.style.color = '#ff4444';
                return;
            }
            
            if (newPass !== confirm) {
                msg.innerHTML = '❌ Mots de passe différents';
                msg.style.color = '#ff4444';
                return;
            }
            
            try {
                await api.request('/auth/change-password', {
                    method: 'POST',
                    body: JSON.stringify({ currentPassword: current, newPassword: newPass })
                });
                
                msg.innerHTML = '✅ Mot de passe modifié';
                msg.style.color = '#0f0';
                setTimeout(() => closeModal('password-modal'), 1500);
            } catch (error) {
                msg.innerHTML = '❌ Erreur: ' + error.message;
                msg.style.color = '#ff4444';
            }
        };

        async function filterUnits(term) {
            if (!token) return;
            try {
                const units = await api.getVests();
                const filtered = units.filter(u => 
                    u.id.toLowerCase().includes(term) || 
                    (u.soldier_name && u.soldier_name.toLowerCase().includes(term))
                );
                
                if (filtered.length < 15 && term === '') {
                    updateUnitsDisplay(units);
                } else {
                    updateUnitsDisplay(filtered);
                }
            } catch (error) {
                console.error('Erreur filtre:', error);
            }
        }
