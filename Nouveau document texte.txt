        function playMilitaryBeep(freq, duration) {
            if (!audioEnabled || !audioCtx || audioCtx.state === 'closed') return;
            if (audioCtx.state === 'suspended') { audioCtx.resume(); return; }
            try {
                const osc = audioCtx.createOscillator();
                const gain = audioCtx.createGain();
                osc.type = 'square';
                osc.frequency.setValueAtTime(freq, audioCtx.currentTime);
                gain.gain.setValueAtTime(0.1, audioCtx.currentTime);
                osc.connect(gain).connect(audioCtx.destination);
                osc.start();
                osc.stop(audioCtx.currentTime + duration);
            } catch(e) {}
        }

async function filterUnits(searchTerm) {

        console.error('❌ Erreur filtrage:', error);

        function updatePositionsOnMap(positions) {
            if (!missionMap) return;
            
            missionMap.eachLayer((layer) => {
                if (layer instanceof L.Marker || layer instanceof L.CircleMarker) {
                    if (!layer.options?.isDrawing) {
                        missionMap.removeLayer(layer);
                    }
                }
            });
            
            positions.forEach(pos => {
                if (pos.latitude && pos.longitude) {
                    const color = pos.status === 'warning' ? '#ff8800' : 
                                 pos.status === 'critical' ? '#ff4444' : '#00ff00';
                    
                    const marker = L.circleMarker([pos.latitude, pos.longitude], {
                        radius: 8,
                        color: color,
                        fillColor: color,
                        fillOpacity: 0.8,
                        weight: 2
                    }).addTo(missionMap);
                    
                    marker.bindPopup(`
                        <b>${pos.id}</b><br>
                        ${pos.soldier_name || ''}<br>
                        📍 ${pos.latitude.toFixed(4)}, ${pos.longitude.toFixed(4)}<br>
                        🕐 ${pos.timestamp ? new Date(pos.timestamp).toLocaleTimeString() : ''}
                    `);
                    
                    marker.on('click', () => showBiometric(pos.id));
                }
            });
        }

        async function loadLatestSensorData() {
            try {
                const vests = await api.getVests();
                const monitoringContent = document.getElementById('monitoring-content');
                
                const latestReadings = [];
                
                for (const vest of vests.slice(0, 5)) {
                    const readings = await api.getSensorReadings(vest.id, 1);
                    if (readings && readings.length > 0) {
                        const reading = readings[0];
                        if (reading.data) {
                            latestReadings.push({
                                sensor_type: `VEST ${vest.id}`,
                                avg_value: reading.data.temperature || reading.data.heart_rate || reading.data.battery || '--',
                                unit: reading.data.temperature ? '°C' : 
                                      reading.data.heart_rate ? 'BPM' : 
                                      reading.data.battery ? '%' : ''
                            });
                        }
                    }
                }
                
                if (latestReadings.length > 0) {
                    displayMonitoringStats(latestReadings);
                } else {
                    displayMonitoringStats(getSimulatedStats());
                }
            } catch (error) {
                console.error('Erreur chargement capteurs:', error);
                displayMonitoringStats(getSimulatedStats());
            }
        }

                    closeModal('password-modal');
                    document.getElementById('current-password').value = '';
                    document.getElementById('new-password').value = '';
                    document.getElementById('confirm-password').value = '';
                    messageEl.innerHTML = '';
                }, 2000);
            } catch (error) {
                messageEl.innerHTML = '❌ Erreur: ' + error.message;

            }
        };

        function initRadar() {
            const canvas = document.getElementById('radar-canvas');
            if (!canvas) return;
            
            const ctx = canvas.getContext('2d');
            let angle = 0;
            
            function resizeCanvas() {
                const container = canvas.parentElement;
                if (container) {
                    canvas.width = container.clientWidth;
                    canvas.height = container.clientHeight;
                }
            }
            
            function draw() {
                resizeCanvas();
                ctx.clearRect(0, 0, canvas.width, canvas.height);
                
                ctx.strokeStyle = 'rgba(0,255,0,0.2)';
                for (let i = 1; i <= 4; i++) {
                    ctx.beginPath();
                    ctx.arc(canvas.width/2, canvas.height/2, i * Math.min(canvas.width, canvas.height)/8, 0, 2*Math.PI);
                    ctx.stroke();
                }
                
                for (let i = 0; i < 8; i++) {
                    ctx.beginPath();
                    ctx.moveTo(canvas.width/2, canvas.height/2);
                    ctx.lineTo(canvas.width/2 + Math.cos(i*Math.PI/4)*canvas.width/2, 
                               canvas.height/2 + Math.sin(i*Math.PI/4)*canvas.height/2);
                    ctx.stroke();
                }
                
                ctx.strokeStyle = '#0f0';
                ctx.lineWidth = 2;
                ctx.beginPath();
                ctx.moveTo(canvas.width/2, canvas.height/2);
                ctx.lineTo(canvas.width/2 + Math.cos(angle)*canvas.width/2, 
                           canvas.height/2 + Math.sin(angle)*canvas.height/2);
                ctx.stroke();
                
                angle += 0.02;
                requestAnimationFrame(draw);
            }
            
            draw();
        }

        async function loadTrends() {
            const unit = document.getElementById('analytics-unit')?.value;

            const interval = document.getElementById('analytics-interval')?.value;

                showNotification('Veuillez sélectionner une unité', 'warning');
                return;

            try {
                const data = await api.getTrends(unit, { interval, range, sensorType: sensor });
                
                if (!trendsChart) {
                    const ctx = document.getElementById('trends-chart')?.getContext('2d');
                    if (!ctx) return;
                    
                    trendsChart = new Chart(ctx, {
                        type: 'line',
                        data: {
                            labels: [],
                            datasets: [{
                                label: `${sensor}`,
                                data: [],
                                borderColor: '#00ff00',
                                backgroundColor: 'rgba(0, 255, 0, 0.1)',
                                tension: 0.4
                            }]
                        },
                        options: {
                            responsive: true,
                            maintainAspectRatio: false,
                            scales: {
                                x: {
                                    type: 'time',
                                    time: { unit: interval?.includes('minute') ? 'minute' : 'hour' },

                                    ticks: { color: '#00ff00' }

                                    grid: { color: 'rgba(0, 255, 0, 0.1)' },

                                legend: { labels: { color: '#00ff00' } }
                            }
                        }
                    });

                if (trendsChart) {
                    trendsChart.data.labels = data.map(d => d.bucket);
                    trendsChart.data.datasets = [{
                        label: `${sensor} - Moyenne`,
                        data: data.map(d => ({ x: d.bucket, y: d.avg_value })),

                        backgroundColor: 'rgba(0, 255, 0, 0.1)',

        async function loadHeatmapChart(unitId, sensorType) {
            if (!unitId) return;
            
            try {
                const data = await api.getTrends(unitId, { 
                    interval: '1 hour', 
                    range: '1 week', 
                    sensorType 
                });
                
                const ctx = document.getElementById('heatmap-chart')?.getContext('2d');
                if (!ctx) return;
                
                if (heatmapChart) heatmapChart.destroy();
                
                const days = ['Dim', 'Lun', 'Mar', 'Mer', 'Jeu', 'Ven', 'Sam'];
                const heatData = [];
                
                data.forEach(d => {
                    const date = new Date(d.bucket);
                    const day = date.getDay();
                    const hour = date.getHours();
                    heatData.push({
                        x: hour,
                        y: day,
                        v: d.avg_value
                    });
                });
                
                heatmapChart = new Chart(ctx, {
                    type: 'scatter',
                    data: {
                        datasets: [{
                            label: sensorType,
                            data: heatData,
                            backgroundColor: (ctx) => {
                                const value = ctx.raw.v;
                                if (!value) return '#000000';
                                const min = 35;
                                const max = 39;
                                const intensity = Math.min(1, Math.max(0, (value - min) / (max - min)));
                                return `rgba(255, ${255 - intensity*200}, 0, 0.8)`;
                            },
                            pointRadius: 15,
                            pointHoverRadius: 18
                        }]
                    },
                    options: {
                        responsive: true,
                        maintainAspectRatio: false,
                        scales: {
                            x: {
                                type: 'linear',
                                position: 'top',
                                min: -0.5,
                                max: 23.5,
                                ticks: {
                                    stepSize: 3,
                                    color: '#00ff00',
                                    callback: (v) => `${v}h`
                                },
                                grid: { color: 'rgba(0,255,0,0.1)' }
                            },
                            y: {
                                type: 'linear',
                                min: -0.5,
                                max: 6.5,
                                ticks: {
                                    stepSize: 1,
                                    color: '#00ff00',
                                    callback: (v) => days[v]
                                },
                                grid: { color: 'rgba(0,255,0,0.1)' }
                            }
                        },
                        plugins: {
                            legend: { display: false },
                            tooltip: {
                                callbacks: {
                                    label: (ctx) => {
                                        const cell = ctx.raw;
                                        const day = days[cell.y];
                                        return `${day} ${cell.x}h: ${cell.v?.toFixed(1)}°C`;
                                    }
                                }
                            }
                        }
                    }
                });
                
            } catch (error) {
                console.error('Erreur heatmap:', error);
            }
        }
